# Преобразование сырого потока в AVI видеофайл. Очень шустро!

ffmpeg -vcodec copy -i ./87-000.out ./87-0000.avi

# Для того чтобы источником служил входной поток (stdin).

ffmpeg -vcodec copy -i - ./87-0000.avi

# Для того чтобы выводило результат не в файл, а в выходной поток (stdout).

ffmpeg -vcodec copy -i ./87-000.out -

# Преобразовать входной поток в картинку (с интервалом в секунду).

ffmpeg -i ./2.frame -r 1 -s 352x288 -f image2 ./pic.jpeg
